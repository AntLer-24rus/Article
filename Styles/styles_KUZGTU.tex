	    \setlength{\listparindent}{\parindent} % Абзацный отступ аннотации

	    \setlength{\rightmargin}{\leftmargin}%
	    \setlength{\parsep}{\parskip}% Отступ ежду абзацами
	  }%
	  \item\relax}
{\endlist}
